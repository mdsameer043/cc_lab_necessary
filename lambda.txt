Experiment 7(basic python program lambda)

import json

def lambda_handler(event, context):
    a = event.get('num1', 0)
    b = event.get('num2', 0)
    result = a + b

    return {
        'statusCode': 200,
        'body': json.dumps({
            'sum': result
        })
    }
